Define a function that simulates a path for $z_t$ given initial conditions and parameters.

In [6]:
function Simulate_z_path(T, , , z0, shock_magnitude)
    d = Normal(0, ) # Create normal distribution object
    _path = rand(d, T) # Draw a vector of shocks
    z_path = zeros(T)
    z_path[round(Int64, T/2)] = shock_magnitude
    
    # Iterate z forward in time
    z_path[1] = *z0 + _path[1]
    for t = 2:T
        z_path[t] += *z_path[t-1] + _path[t]
    end
    
    return z_path
end

function Simulate_system(v_ck, v_cz, v_kk, v_kz, k0, T, , , z0, shock_magnitude)
    c_path = zeros(T)
    k_path = zeros(T)
    i_path = zeros(T)
    y_path = zeros(T)
    w_path = zeros(T)
    z_path = Simulate_z_path(T, , , z0, shock_magnitude)
    
    for t = 2:T
        k_path[t] = v_kk*k_path[t-1] + v_kz*z_path[t]
        c_path[t] = v_ck*k_path[t-1] + v_cz*z_path[t]
        i_path[t] = (k_path[t] - (1-)*k_path[t-1])/
        y_path[t] = z_path[t] + *k_path[t-1]
        w_path[t] = y_path[t]
    end
    
    return (k_path, c_path, i_path, y_path, w_path, z_path)
end
